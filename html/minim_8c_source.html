139 void adjust_alpha(const int natom, const double grad_old[], const double grad_new[], double *alpha)
140 {
141  double dot_prod=0.0;
142  double norm_old=0.0;
143  double norm_new=0.0;
144  double angle;
145 
146  int i;
147 
148  for(i=0; i<natom; i++)
149  {
150  dot_prod += grad_old[i]*grad_new[i];
151  norm_old += X2(grad_old[i]);
152  norm_new += X2(grad_new[i]);
153  }
154  norm_old = sqrt(norm_old);
155  norm_new = sqrt(norm_new);
156 
157  angle=acos(dot_prod/(norm_old*norm_new));
158  angle*=180.0/3.14159265358979323846;
159 
160  if (angle > 60.0)
161  *alpha *= 0.5;
162  else
163  *alpha *= 1.05;
164 }
